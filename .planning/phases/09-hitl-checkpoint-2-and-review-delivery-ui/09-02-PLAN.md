From 09-RESEARCH.md (Google Drive iframe URLs):
```typescript
// Slides embed: https://docs.google.com/presentation/d/{ID}/embed?start=false&loop=false&delayms=3000
// Docs embed: https://docs.google.com/document/d/{ID}/preview
// Edit URL: use driveUrl as-is (already ends in /edit)
function getEmbedUrl(driveUrl: string, type: "slides" | "docs"): string {
  const id = driveUrl.split("/d/")[1]?.split("/")[0];
  if (!id) return driveUrl;
  if (type === "slides") return `https://docs.google.com/presentation/d/${id}/embed?start=false&loop=false&delayms=3000`;
  return `https://docs.google.com/document/d/${id}/preview`;
}
```
