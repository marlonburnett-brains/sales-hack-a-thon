**1. Add Action Required API helpers to api-client.ts:**

Add these exported functions at the end of the file (use `fetchJSON`, not `fetchWithGoogleAuth` -- these are CRUD operations, no Google API involved):

```typescript
// ── Action Required ──

export interface ActionRequiredItem {
  id: string;
  userId: string | null;
  actionType: string;
  title: string;
  description: string;
  resourceId: string | null;
  resourceName: string | null;
  resolved: boolean;
  resolvedAt: string | null;
  createdAt: string;
  updatedAt: string;
}

export async function fetchActions(): Promise<ActionRequiredItem[]> {
  return fetchJSON<ActionRequiredItem[]>("/actions");
}

export async function fetchActionCount(): Promise<number> {
  const result = await fetchJSON<{ count: number }>("/actions/count");
  return result.count;
}

export async function resolveAction(id: string): Promise<ActionRequiredItem> {
  return fetchJSON<ActionRequiredItem>(`/actions/${id}/resolve`, {
    method: "PATCH",
  });
}
```

**2. Add badge count for Action Required:**

The sidebar is a client component ("use client"). Add a `pendingCount` state and fetch it on mount:

```typescript
const [pendingCount, setPendingCount] = useState(0);

useEffect(() => {
  // Fetch action count from agent API via a lightweight endpoint
  fetch("/api/actions/count")
    .then((res) => res.json())
    .then((data) => setPendingCount(data.count ?? 0))
    .catch(() => {}); // silent fail
}, [pathname]); // re-fetch on navigation (catches resolves)
```

Create a Next.js API route at `apps/web/src/app/(authenticated)/api/actions/count/route.ts` that proxies to the agent:
```typescript
import { fetchActionCount } from "@/lib/api-client";
import { NextResponse } from "next/server";

export async function GET() {
  try {
    const count = await fetchActionCount();
    return NextResponse.json({ count });
  } catch {
    return NextResponse.json({ count: 0 });
  }
}
```
